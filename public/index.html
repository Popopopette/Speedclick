// script.js const socket = io(); const canvas = document.getElementById("gameCanvas"); const ctx = canvas.getContext("2d"); const chatForm = document.getElementById("chatForm"); const chatInput = document.getElementById("chatInput"); const chatDiv = document.getElementById("chat"); const lobby = document.getElementById("lobby"); const startBtn = document.getElementById("startBtn"); const playersList = document.getElementById("playersList"); const timerDiv = document.getElementById("timer"); const leaderboardDiv = document.getElementById("leaderboard"); let pseudo = prompt("Entrez votre pseudo :") || "Anonyme"; let players = []; let currentShape = null; let clickEffects = []; let lastMouseX = 0; let lastMouseY = 0; const clickSound = new Audio("Impact_Speedclick.mp3"); socket.emit("setPseudo", pseudo); socket.on("lobbyUpdate", ({ players: updatedPlayers, hostId }) => { players = updatedPlayers; playersList.innerHTML = "

Joueurs connect√©s :

" + players.map(p => `
${p.pseudo}
`).join(""); if (socket.id === hostId) startBtn.style.display = "inline-block"; lobby.style.display = "block"; }); startBtn.onclick = () => { socket.emit("startGame"); lobby.style.display = "none"; canvas.style.display = "block"; }; socket.on("startRound", shape => { currentShape = shape; timerDiv.style.display = "block"; }); socket.on("updateLeaderboard", leaderboard => { leaderboardDiv.style.display = "block"; leaderboardDiv.innerHTML = `

Classement

` + leaderboard.map((p, i) => `
${i + 1}. ${p.pseudo} - ${p.score} pts
`).join(""); }); socket.on("clickAccepted", () => { clickSound.currentTime = 0; clickSound.play(); clickEffects.push({ x: lastMouseX, y: lastMouseY, radius: 10, alpha: 1 }); }); canvas.addEventListener("mousemove", e => { const rect = canvas.getBoundingClientRect(); lastMouseX = e.clientX - rect.left; lastMouseY = e.clientY - rect.top; socket.emit("mouseMove", { x: lastMouseX, y: lastMouseY }); }); canvas.addEventListener("click", e => { if (!currentShape) return; const rect = canvas.getBoundingClientRect(); const x = e.clientX - rect.left; const y = e.clientY - rect.top; const dx = x - currentShape.x; const dy = y - currentShape.y; const inShape = dx * dx + dy * dy <= currentShape.size * currentShape.size; if (inShape) { socket.emit("playerClick"); } }); function drawEverything() { ctx.clearRect(0, 0, canvas.width, canvas.height); if (currentShape) { ctx.beginPath(); ctx.fillStyle = currentShape.color; ctx.arc(currentShape.x, currentShape.y, currentShape.size, 0, Math.PI * 2); ctx.fill(); } // Dessiner les effets de clic clickEffects.forEach((fx, index) => { ctx.beginPath(); ctx.strokeStyle = `rgba(0, 0, 0, ${fx.alpha})`; ctx.arc(fx.x, fx.y, fx.radius, 0, Math.PI * 2); ctx.stroke(); fx.radius += 2; fx.alpha -= 0.05; if (fx.alpha <= 0) clickEffects.splice(index, 1); }); requestAnimationFrame(drawEverything); } drawEverything(); chatForm.addEventListener("submit", e => { e.preventDefault(); if (chatInput.value) { socket.emit("chatMessage", chatInput.value); chatInput.value = ""; } }); socket.on("chatMessage", ({ pseudo, msg }) => { chatDiv.innerHTML += `
${pseudo}: ${msg}
`; chatDiv.scrollTop = chatDiv.scrollHeight; });
